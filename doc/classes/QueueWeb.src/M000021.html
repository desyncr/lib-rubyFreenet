  def do_POST(req, res)
    case req.path
    when '/add_uris'
      params = CGI::parse(req.body)
      params['uris'].each do |uris|
        @drb.add_uris(*(uris.split(/\r|\n/)))
      end
      res.status = 302
      res['Location'] = '/'
    else
    end
  end
