      def generate_keypair(async = false, &callback)
        message = Message.new('GenerateSSK', nil, nil, callback)
        send(message, async)
        if async
          message
        else
          message.wait_for_response
          [message.response.items['InsertURI'], message.response.items['RequestURI']]
        end
      end
