      def connect
        @lock.synchronize do
          log(INFO, "Connecting to #{@server}:#{@port}")
          @socket = TCPSocket.open(@server, @port)
          @running = true
          @socket_thread = Thread.new {socket_thread}
          log(DEBUG, "Running thread")
        end
      rescue
        raise FCPConnectionError.new('Connection failed')
      end
