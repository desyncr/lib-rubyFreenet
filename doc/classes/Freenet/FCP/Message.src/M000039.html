      def write(stream)
        stream.write(type.strip+"\n")
        items.each do |key, value|
          stream.write("#{key}=#{value.to_s.strip}\n")
        end
        if data
          stream.write("DataLength=#{data.length}\n")
          stream.write("Data\n")
          stream.write(data)
        else
          stream.write("EndMessage\n")
        end
        stream.flush
      end
