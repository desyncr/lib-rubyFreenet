    def merge(uri)
      if uri =~ /^\//
        return uri
      end
      uri = uri.strip
      begin
        uri = URI.new(uri) unless uri.respond_to? :uri
        if uri.site == @site
          return "#{@site}#{merge_paths(@path, uri.path)}"
        else
          return uri.uri
        end
      rescue URIError => e # We have a fragment
        "#{@site}#{merge_paths(@path, uri)}"
      end
    end
