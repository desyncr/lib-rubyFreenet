    def uri
      case @type
      when 'KSK'
        "#{@type}@#{@key}"
      when 'CHK'
        "#{@type}@#{@key}/#{@path}"
      when 'USK'
        "#{@type}@#{@key}/#{@name}/#{@version.to_s}/#{@path}#{"?#{@query_string}" if @query_string}#{"##{@anchor}" if @anchor}"
      when 'SSK'
        "#{@type}@#{@key}/#{@name}#{'-'+@version.to_s if @version}/#{@path}#{"?#{@query_string}" if @query_string}#{"##{@anchor}" if @anchor}"
      end
    end
